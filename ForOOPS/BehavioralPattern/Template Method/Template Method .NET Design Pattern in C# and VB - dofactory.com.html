Template Method


 Definition
 UML diagram
 Participants
 Structural code in C#
 Real-world code in C#
 .NET Optimized code in C#


Definition

Define the skeleton of an algorithm in an operation, deferring some steps to subclasses. Template Method lets subclasses redefine certain steps of an algorithm without changing the algorithm's structure.

Frequency of use:
Medium



UML class diagram





Participants


    The classes and objects participating in this pattern are:

  • AbstractClass  (DataObject)
    • defines abstract primitive operations that concrete subclasses define to implement steps of an algorithm
    • implements a template method defining the skeleton of an algorithm. The template method calls primitive operations as well as operations defined in AbstractClass or those of other objects.
  • ConcreteClass  (CustomerDataObject)
    • implements the primitive operations ot carry out subclass-specific steps of the algorithm


Structural code in C#


This structural code demonstrates the Template method which provides a skeleton calling sequence of methods. One or more steps can be deferred to subclasses which implement these steps without changing the overall calling sequence. 

  1.                  
  3. using System;
  4.  
  5. namespace DoFactory.GangOfFour.Template.Structural
  6. {
  7.   /// <summary>
  8.   /// MainApp startup class for Real-World 
  9.   /// Template Design Pattern.
  10.   /// </summary>
  11.   class MainApp
  12.   {
  13.     /// <summary>
  14.     /// Entry point into console application.
  15.     /// </summary>
  16.     static void Main()
  17.     {
  18.       AbstractClass aA = new ConcreteClassA();
  19.       aA.TemplateMethod();
  20.  
  21.       AbstractClass aB = new ConcreteClassB();
  22.       aB.TemplateMethod();
  23.  
  24.       // Wait for user
  25.       Console.ReadKey();
  26.     }
  27.   }
  28.  
  29.   /// <summary>
  30.   /// The 'AbstractClass' abstract class
  31.   /// </summary>
  32.   abstract class AbstractClass
  33.   {
  34.     public abstract void PrimitiveOperation1();
  35.     public abstract void PrimitiveOperation2();
  36.  
  37.     // The "Template method"
  38.     public void TemplateMethod()
  39.     {
  40.       PrimitiveOperation1();
  41.       PrimitiveOperation2();
  42.       Console.WriteLine("");
  43.     }
  44.   }
  45.  
  46.   /// <summary>
  47.   /// A 'ConcreteClass' class
  48.   /// </summary>
  49.   class ConcreteClassA : AbstractClass
  50.   {
  51.     public override void PrimitiveOperation1()
  52.     {
  53.       Console.WriteLine("ConcreteClassA.PrimitiveOperation1()");
  54.     }
  55.     public override void PrimitiveOperation2()
  56.     {
  57.       Console.WriteLine("ConcreteClassA.PrimitiveOperation2()");
  58.     }
  59.   }
  60.  
  61.   /// <summary>
  62.   /// A 'ConcreteClass' class
  63.   /// </summary>
  64.   class ConcreteClassB : AbstractClass
  65.   {
  66.     public override void PrimitiveOperation1()
  67.     {
  68.       Console.WriteLine("ConcreteClassB.PrimitiveOperation1()");
  69.     }
  70.     public override void PrimitiveOperation2()
  71.     {
  72.       Console.WriteLine("ConcreteClassB.PrimitiveOperation2()");
  73.     }
  74.   }
  75. }

Real-world code in C#


This real-world code demonstrates a Template method named Run() which provides a skeleton calling sequence of methods. Implementation of these steps are deferred to the CustomerDataObject subclass which implements the Connect, Select, Process, and Disconnect methods. 

  1.                  
  3. using System;
  4. using System.Data;
  5. using System.Data.OleDb;
  6.  
  7. namespace DoFactory.GangOfFour.Template.RealWorld
  8. {
  9.   /// <summary>
  10.   /// MainApp startup class for Real-World 
  11.   /// Template Design Pattern.
  12.   /// </summary>
  13.   class MainApp
  14.   {
  15.     /// <summary>
  16.     /// Entry point into console application.
  17.     /// </summary>
  18.     static void Main()
  19.     {
  20.       DataAccessObject daoCategories = new Categories();
  21.       daoCategories.Run();
  22.  
  23.       DataAccessObject daoProducts = new Products();
  24.       daoProducts.Run();
  25.  
  26.       // Wait for user
  27.       Console.ReadKey();
  28.     }
  29.   }
  30.  
  31.   /// <summary>
  32.   /// The 'AbstractClass' abstract class
  33.   /// </summary>
  34.   abstract class DataAccessObject
  35.   {
  36.     protected string connectionString;
  37.     protected DataSet dataSet;
  38.  
  39.     public virtual void Connect()
  40.     {
  41.       // Make sure mdb is available to app
  42.       connectionString =
  43.         "provider=Microsoft.JET.OLEDB.4.0; " +
  44.         "data source=..\\..\\..\\db1.mdb";
  45.     }
  46.  
  47.     public abstract void Select();
  48.     public abstract void Process();
  49.  
  50.     public virtual void Disconnect()
  51.     {
  52.       connectionString = "";
  53.     }
  54.  
  55.     // The 'Template Method' 
  56.     public void Run()
  57.     {
  58.       Connect();
  59.       Select();
  60.       Process();
  61.       Disconnect();
  62.     }
  63.   }
  64.  
  65.   /// <summary>
  66.   /// A 'ConcreteClass' class
  67.   /// </summary>
  68.   class Categories : DataAccessObject
  69.   {
  70.     public override void Select()
  71.     {
  72.       string sql = "select CategoryName from Categories";
  73.       OleDbDataAdapter dataAdapter = new OleDbDataAdapter(
  74.         sql, connectionString);
  75.  
  76.       dataSet = new DataSet();
  77.       dataAdapter.Fill(dataSet, "Categories");
  78.     }
  79.  
  80.     public override void Process()
  81.     {
  82.       Console.WriteLine("Categories ---- ");
  83.  
  84.       DataTable dataTable = dataSet.Tables["Categories"];
  85.       foreach (DataRow row in dataTable.Rows)
  86.       {
  87.         Console.WriteLine(row["CategoryName"]);
  88.       }
  89.       Console.WriteLine();
  90.     }
  91.   }
  92.  
  93.   /// <summary>
  94.   /// A 'ConcreteClass' class
  95.   /// </summary>
  96.   class Products : DataAccessObject
  97.   {
  98.     public override void Select()
  99.     {
  100.       string sql = "select ProductName from Products";
  101.       OleDbDataAdapter dataAdapter = new OleDbDataAdapter(
  102.         sql, connectionString);
  103.  
  104.       dataSet = new DataSet();
  105.       dataAdapter.Fill(dataSet, "Products");
  106.     }
  107.  
  108.     public override void Process()
  109.     {
  110.       Console.WriteLine("Products ---- ");
  111.       DataTable dataTable = dataSet.Tables["Products"];
  112.       foreach (DataRow row in dataTable.Rows)
  113.       {
  114.         Console.WriteLine(row["ProductName"]);
  115.       }
  116.       Console.WriteLine();
  117.     }
  118.   }
  119. }
  120.
